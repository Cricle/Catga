💻 完整使用示例

1️⃣ 基础命令处理

// 定义命令
public record CreateOrderCommand(
    string ProductId,
    int Quantity
) : IRequest<Order>;

// 实现 Handler（自动注册，无需配置）
public class CreateOrderHandler : IRequestHandler<CreateOrderCommand, Order>
{
    private readonly IOrderRepository _repository;

    public CreateOrderHandler(IOrderRepository repository)
    {
        _repository = repository;
    }

    public async Task<CatgaResult<Order>> HandleAsync(
        CreateOrderCommand request,
        CancellationToken cancellationToken)
    {
        var order = new Order(request.ProductId, request.Quantity);
        await _repository.SaveAsync(order, cancellationToken);
        return CatgaResult<Order>.Success(order);
    }
}

// 使用
var result = await mediator.SendAsync<CreateOrderCommand, Order>(
    new CreateOrderCommand("product-123", 5)
);

if (result.IsSuccess)
{
    Console.WriteLine($"✅ 订单已创建: {result.Value.Id}");
}

2️⃣ 完整的异常处理

// Handler 中的异常处理
public async Task<CatgaResult<Order>> HandleAsync(
    CreateOrderCommand request,
    CancellationToken cancellationToken)
{
    // 参数验证
    if (string.IsNullOrEmpty(request.ProductId))
        return CatgaResult<Order>.Failure("产品ID不能为空");

    if (request.Quantity <= 0)
        return CatgaResult<Order>.Failure("数量必须大于0");

    try
    {
        // 检查产品是否存在
        var product = await _productRepo.GetByIdAsync(request.ProductId, cancellationToken);
        if (product == null)
            return CatgaResult<Order>.Failure($"产品不存在: {request.ProductId}");

        // 检查库存
        if (product.Stock < request.Quantity)
            return CatgaResult<Order>.Failure($"库存不足，剩余: {product.Stock}");

        // 创建订单
        var order = new Order(request.ProductId, request.Quantity);
        await _orderRepo.SaveAsync(order, cancellationToken);

        return CatgaResult<Order>.Success(order);
    }
    catch (DbUpdateException ex)
    {
        return CatgaResult<Order>.Failure($"数据库错误: {ex.Message}");
    }
    catch (OperationCanceledException)
    {
        return CatgaResult<Order>.Failure("操作已取消");
    }
    catch (Exception ex)
    {
        // 记录日志
        _logger.LogError(ex, "创建订单失败");
        return CatgaResult<Order>.Failure($"未知错误: {ex.Message}");
    }
}

// 调用端完整异常处理
try
{
    var result = await mediator.SendAsync<CreateOrderCommand, Order>(
        new CreateOrderCommand("product-123", 5),
        cancellationToken
    );

    if (result.IsSuccess)
    {
        // 成功处理
        Console.WriteLine($"✅ 订单创建成功: {result.Value.Id}");
        return Results.Ok(result.Value);
    }
    else
    {
        // 业务失败
        Console.WriteLine($"❌ 失败: {result.Error}");
        return Results.BadRequest(new { error = result.Error });
    }
}
catch (TimeoutException ex)
{
    // 超时异常
    _logger.LogWarning(ex, "命令执行超时");
    return Results.StatusCode(504);
}
catch (OperationCanceledException)
{
    // 取消异常
    _logger.LogInformation("用户取消操作");
    return Results.StatusCode(499);
}
catch (Exception ex)
{
    // 未预期异常
    _logger.LogError(ex, "命令执行失败");
    return Results.StatusCode(500);
}

3️⃣ 事件发布与订阅（带异常处理）

// 定义事件
public record OrderCreatedEvent(
    string OrderId,
    string ProductId,
    int Quantity,
    DateTime CreatedAt
) : IEvent;

// 发布事件（Handler 内）
var order = new Order(request.ProductId, request.Quantity);
await _orderRepo.SaveAsync(order, cancellationToken);

// 发布事件到消息总线
var @event = new OrderCreatedEvent(
    order.Id, order.ProductId, order.Quantity, DateTime.UtcNow
);
await mediator.PublishAsync(@event, cancellationToken);

// 事件 Handler 1: 发送通知（自动注册）
public class SendEmailHandler : IEventHandler<OrderCreatedEvent>
{
    private readonly IEmailService _emailService;

    public async Task HandleAsync(
        OrderCreatedEvent @event,
        CancellationToken cancellationToken)
    {
        try
        {
            await _emailService.SendAsync(
                "order-confirmation.html",
                new { @event.OrderId, @event.ProductId },
                cancellationToken
            );
        }
        catch (Exception ex)
        {
            // 记录日志但不抛出（避免影响其他 Handler）
            _logger.LogError(ex, "发送邮件失败: {OrderId}", @event.OrderId);
        }
    }
}

// 事件 Handler 2: 更新库存（自动注册）
public class UpdateInventoryHandler : IEventHandler<OrderCreatedEvent>
{
    private readonly IInventoryService _inventoryService;

    public async Task HandleAsync(
        OrderCreatedEvent @event,
        CancellationToken cancellationToken)
    {
        await _inventoryService.DecreaseStockAsync(
            @event.ProductId,
            @event.Quantity,
            cancellationToken
        );
    }
}

4️⃣ 完整配置示例

// Program.cs - 生产环境完整配置
var builder = WebApplication.CreateBuilder(args);

// 添加 Catga 核心
builder.Services
    .AddCatga()
    .AddGeneratedHandlers();  // Source Generator 自动注册所有 Handler

// 添加 Redis 传输层（支持 QoS 0 Pub/Sub 和 QoS 1 Streams）
builder.Services.AddRedisTransport(options =>
{
    options.Configuration = builder.Configuration.GetConnectionString("Redis");
    options.DefaultQoS = QoS.AtLeastOnce;  // QoS 0 或 QoS 1
    options.StreamMaxLength = 10000;       // Stream 最大长度
});

// 添加 Redis 持久化（Outbox/Inbox）
builder.Services.AddRedisOutbox(options =>
{
    options.Configuration = builder.Configuration.GetConnectionString("Redis");
    options.BatchSize = 100;               // 批量发送
    options.PollingInterval = TimeSpan.FromSeconds(5);
});

builder.Services.AddRedisInbox(options =>
{
    options.Configuration = builder.Configuration.GetConnectionString("Redis");
    options.RetentionPeriod = TimeSpan.FromDays(7);  // 保留 7 天
});

// 添加 OpenTelemetry 监控（可选）
builder.Services.AddOpenTelemetry()
    .WithTracing(tracing => tracing
        .AddSource("Catga.Framework")  // Catga 追踪
        .AddAspNetCoreInstrumentation()
        .AddConsoleExporter())
    .WithMetrics(metrics => metrics
        .AddMeter("Catga.Framework")   // Catga 指标
        .AddAspNetCoreInstrumentation()
        .AddConsoleExporter());

// 或使用 NATS（替代 Redis）
// builder.Services.AddNatsTransport(options => ...
// builder.Services.AddNatsOutbox(options => ...

// 或使用内存（开发环境）
// builder.Services.AddInMemoryTransport();
// builder.Services.AddInMemoryOutbox();

var app = builder.Build();
app.Run();

5️⃣ Minimal API 集成

// 完整的 API 端点（含异常处理）
app.MapPost("/orders", async (
    CreateOrderRequest request,
    ICatgaMediator mediator,
    ILogger<Program> logger,
    CancellationToken cancellationToken) =>
{
    try
    {
        // 发送命令
        var command = new CreateOrderCommand(request.ProductId, request.Quantity);
        var result = await mediator.SendAsync<CreateOrderCommand, Order>(
            command,
            cancellationToken
        );

        // 检查结果
        if (result.IsSuccess)
        {
            logger.LogInformation("订单创建成功: {OrderId}", result.Value.Id);
            return Results.Created($"/orders/{result.Value.Id}", result.Value);
        }
        else
        {
            logger.LogWarning("订单创建失败: {Error}", result.Error);
            return Results.BadRequest(new { error = result.Error });
        }
    }
    catch (OperationCanceledException)
    {
        logger.LogInformation("请求已取消");
        return Results.StatusCode(499);
    }
    catch (Exception ex)
    {
        logger.LogError(ex, "处理订单请求时发生错误");
        return Results.Problem("服务器内部错误");
    }
})
.WithName("CreateOrder")
.WithOpenApi();

🚀 快速开始

安装 Catga（3 步上手）

1️⃣ 安装 NuGet 包

# 核心包（必需）
dotnet add package Catga

# 传输层（三选一）
dotnet add package Catga.Transport.InMemory   # 开发环境
dotnet add package Catga.Transport.Redis      # 生产环境
dotnet add package Catga.Transport.Nats       # 生产环境

# 持久化（可选，选一个）
dotnet add package Catga.Persistence.Redis    # Outbox/Inbox
dotnet add package Catga.Persistence.Nats     # Outbox/Inbox

# Source Generator（推荐，自动注册 Handler）
dotnet add package Catga.SourceGenerator

2️⃣ 配置服务（Program.cs）

var builder = WebApplication.CreateBuilder(args);

// 添加 Catga（一行代码）
builder.Services
    .AddCatga()
    .AddGeneratedHandlers()      // 自动注册
    .AddInMemoryTransport();     // 或 AddRedisTransport()

var app = builder.Build();
app.Run();

3️⃣ 使用（发送命令）

// 注入 Mediator
var mediator = app.Services.GetRequiredService<ICatgaMediator>();

// 发送命令
var result = await mediator.SendAsync<MyCommand, MyResult>(
    new MyCommand()
);

// 检查结果
if (result.IsSuccess)
    Console.WriteLine("成功！");
else
    Console.WriteLine($"失败: {result.Error}");
